-- Create ENUM type for status
CREATE TYPE work_order_status AS ENUM ('pending', 'completed');

-- Create work_order table
CREATE TABLE work_order (
    workorder_id SERIAL PRIMARY KEY,
    tower_id INTEGER REFERENCES towers(tower_id),
    technician_id INTEGER REFERENCES technicians(technician_id),
    task_description VARCHAR(255),
    scheduled_date DATE,
    status work_order_status,
    completed_date DATE,
    created_date DATE DEFAULT CURRENT_DATE,
    deleted_status BOOLEAN DEFAULT FALSE
);
